For the Curious

Filters you can actually understand

Every filter in ByteWave shows you its shader code. Edit it directly, or copy the reference and use AI to create something entirely new.

// 🎨 Impressionist background filter
// Copy this to ChatGPT: "Make it more like Monet with warmer tones"

uniform sampler2D inputImage;
uniform sampler2D subjectMask;
uniform float brushSize = 4.0;

vec4 main(vec2 uv) {
  float mask = texture(subjectMask, uv).r;
  vec4 color = impressionist(inputImage, uv, brushSize);
  return mix(color, texture(inputImage, uv), mask);
}
